    class SiegeState {
  constructor () {
    this.canMove = false;
    this.damage = 20;
  }
}
class TankState {
  constructor () {
    this.canMove = true;
    this.damage = 5;
  }
}
class Tank {
  constructor () {
    this.state = new TankState();
  }
  set state (state) {
    this.state = state;
  }
  get canMove () {
    return this.state.canMove;
  }
  get damage () {
    return this.state.damage;
  }
  onSendingMessage (message) {
    switch (message) {
      case 'Tank':
        this.state = new TankState();
        break;
      case 'Siege':
        this.state = new SiegeState();
        break;
    }
  }
}
